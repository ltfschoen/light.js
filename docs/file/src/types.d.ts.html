export type Metadata = {
  calledWithArgs?: {
    [key: string]: any;
  };
  calls?: string[];
  dependsOn?: string[];
  frequency?: Observable<any>[];
  name?: string;
};

export interface FrequencyObservable<T> extends Observable<T> {
  metadata: { name: string };
}

export interface RpcObservable<T> {

}
